```
pip install django
```
```
pip install djangorestframework
```

```
pip install Pillow
```
```
pip install django-filter
```
```
pip install django-cors-headers
```
```
pip install djangorestframework-simplejwt
```
- For frontend
```
npm install
```
```
npm start
```
